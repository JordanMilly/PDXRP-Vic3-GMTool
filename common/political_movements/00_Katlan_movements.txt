movement_katla_schizo = {
	category = movement_category_ideological
	
	# Nihilists pretty much only care about their atheist thing
	law_enactment_radicalism_multiplier = 1
	
	ideology = ideology_katla_schizo
	character_ideologies = {
		ideology_katla_schizo
	}
	
	disband_trigger = {
		always = no
	}
	
	on_disbanded = {
	}
	
	creation_trigger = {
		has_modifier = winter_of_the_world
	}	
	creation_weight = {
		value = 100
		if = {
			limit = {
				owner ?= {
					any_neighbouring_state = {
						owner = {
							any_political_movement = {
								is_political_movement_type = movement_katla_schizo
							}
						}
					}
				}
			}
			add = {
				value = 100
			}
		}
		if = {
			limit = {
				owner ?= {
					any_neighbouring_state = {
						owner = {
							OR = {
								has_law = law_type:law_anarchy
								ruler = {
									has_ideology = ideology:ideology_katla_schizo
								}
							}
						}
					}
				}
			}
			add = {
				value = 100
			}
		}
		multiply = { 
			add = owner.radical_fraction
			multiply = 0.80
			add = 1
		}
	}	
	on_created = {
		add_movement_enthusiasm_modifier = yes
	}
	on_disbanded = {
	}
	
	character_support_trigger = {
		OR = {
			has_ideology = ideology:ideology_katla_schizo
			has_ideology = ideology:ideology_atheist
			has_ideology = ideology:ideology_positivist
			has_ideology = ideology:ideology_anarchist
		}
	}
	character_support_weight = {
		value = 200
		if = {
			limit = {
				has_ideology = ideology:ideology_katla_schizo
			}
			multiply = {
				value = 5
			}
		}
	}

	pop_support_trigger = {
		always = yes
	}

	pop_support_factors = {
		movement_support_low_literacy
		movement_support_radicalization
	}
	
	pop_support_weight = {	
		add = {
			value = 1
			desc = "POP_BASE_SUPPORT"
		}
		
		add = { # Multiplied by 2x max
			desc = "POP_RADICALS"
			value = pop_radical_fraction
			multiply = 5
		}
		if = {
			limit = {
				owner ?= {
					any_neighbouring_state = {
						owner = {
							any_political_movement = {
								is_political_movement_type = movement_katla_schizo
							}
						}
					}
				}
			}
			add = {
				desc = "Katla_cult_in_neigbor"
				value = 5
			}
		}
		if = {
			limit = {
				owner ?= {
					ruler = {
						has_ideology = ideology:ideology_katla_schizo
					}
				}
			}
			add = {
				desc = "Katla_ruler"
				value = 20
			}
		}
		
		if = {
			limit = {
				owner ?= {
					has_law = law_type:law_anarchy
				}
			}
			add = {
				desc = "Anarchy"
				value = 5
			}
		}
		
		if = {
			limit = {
				owner ?= {
					any_neighbouring_state = {
						owner = {
							OR = {
								has_law = law_type:law_anarchy
								ruler = {
									has_ideology = ideology:ideology_katla_schizo
								}
							}
						}
					}
				}
			}
			add = {
				desc = "Katla_cult_neigbor"
				value = 5
			}
		}
		multiply = { # Illiterate pops are more likely to join
			desc = "POP_LITERACY"
			value = literacy_rate
			multiply = -1
			add = 2
		}
		
		
		if = {
			limit = {
				owner ?= {
					ruler = {
						has_ideology = ideology:ideology_katla_schizo
					}
				}
			}
			multiply = {
				desc = "Katla_ruler"
				value = pop_loyalist_fraction
				add = 1
			}
		}
		
		multiply = { # Multiplied by 3x max
			desc = "POP_RADICALS"
			value = pop_radical_fraction
			multiply = 2
			add = 1
		}
	}
	
	revolution = {
		possible = {
			political_movement_support >= 0.05
		}
		
		weight = {
			value = 100
		}	
	
		state_weight = {
			value = radical_fraction
			multiply = 0.8
			if = {
				limit = { is_incorporated = yes }
				add = 0.3
			}
			add = {
				value = literacy_rate
				multiply = -0.9
			}
			if = {
				limit = { is_capital = yes }
				multiply = 0
			}
		}
	}
	
	secession = {
		possible = {
			political_movement_support >= 0.05
			scope:clout <= 0
		}
		
		weight = {
			value = 100
		}
	
		state_weight = {
			value = radical_fraction
			multiply = 0.8
			if = {
				limit = { is_incorporated = yes }
				add = 0.3
			}
			add = {
				value = literacy_rate
				multiply = -0.9
			}
			if = {
				limit = { is_capital = yes }
				multiply = 0
			}
		}		
	}	
}
